;初始化日常活动状态
@Gosub System/System_Subroutine.BeginAdv

;显示背景
@back 5_1 Scale:{g_backgroundDefaultScale} Lazy:false
;最上面的黑色遮罩层VerticalBlindGradually特效消失
@back SubId:"Overlay" Transparent.Custom dissolve:"VerticalBlindGradually" tint:"#000000" time:0.99 Lazy:false
;等待一小会到特效演完
@Wait "0.9"

;显示艾玛，表情是Default，双手是状态1
@char Ema.Default,ArmR1,ArmL1 pos:50,50 Scale:1.25 Lazy:false
;艾玛说话
Ema: 这是一个测试用例，我来客串一下。
Ema: 再说一句就下线了。
;隐藏艾玛
@hide Ema Lazy:false
