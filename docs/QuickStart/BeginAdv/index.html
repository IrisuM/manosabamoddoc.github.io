          1. Suspendisse sed congue orci. #

            
;初始化日常活动状态
@Gosub System/System_Subroutine.BeginAdv

;显示背景
@back 5_1 Scale:{g_backgroundDefaultScale} Lazy:false
;最上面的黑色遮罩层VerticalBlindGradually特效消失
@back SubId:"Overlay" Transparent.Custom dissolve:"VerticalBlindGradually" tint:"#000000" time:0.99 Lazy:false
;等待一小会到特效演完
@Wait "0.9"

;显示艾玛，表情是Default，双手是状态1
@char Ema.Default,ArmR1,ArmL1 pos:50,50 Scale:1.25 Lazy:false
;艾玛说话
Ema: 这是一个测试用例，我来客串一下。
Ema: 再说一句就下线了。
;隐藏艾玛
@hide Ema Lazy:false

这里是一段空白。

;显示雪莉，表情是Smile1，双手是状态1
@char Sherry.Smile1,ArmR1,ArmL1 pos:30,50 Scale:1.25 Lazy:false
;雪莉说话
Sherry: 这是一个测试用例，我也来客串一下。
Sherry: 接下来请汉娜出场。
;显示雪莉，表情是Smile1，双手是状态2
@char Hanna.Angry1,ArmR2,ArmL2 pos:70,50 Lazy:false
;雪莉说话
Hanna: 没事不要随便叫我出来啦。
;隐藏全部角色
@hideChars Lazy:false

;结束化日常活动状态
@Gosub System/System_Subroutine.EndAdv

;跳转到结束
@goto SherryAppleJuice_ExampleMod/Tools.EndToTitle
